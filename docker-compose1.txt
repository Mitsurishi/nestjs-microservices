version: '3'

services:

  postgres_auth:
    image: postgres
    env_file:
      - .auth.env
    ports:
      - '5432:5432'
    volumes:
      - ./db/auth-db:/var/lib/postgresql/data
    networks:
      - postgres_auth

  postgres_profile:
    image: postgres
    env_file:
      - .profile.env
    ports:
      - '5433:5432'
    volumes:
      - ./db/profile-db:/var/lib/postgresql/data
    networks:
      - postgres_profile

  postgres_admin:
    image: dpage/pgadmin4
    depends_on:
      - postgres_auth
      - postgres_profile
    env_file:
      - .env
    ports:
      - '15432:80'
    volumes:
      - ./db/pgadmin:/var/lib/data
    networks:
      - postgres_auth
      - postgres_profile

  rabbitmq:
    image: rabbitmq:3-management
    container_name: rabbitmq
    volumes:
      - /var/lib/rabbitmq
    ports:
      - '5672:5672'
      - '15672:15672'
    env_file:
      - .env
    networks:
      - rabbitmq

  auth:
    build:
      context: ./
      dockerfile: ./apps/auth/Dockerfile
    env_file:
      - ./apps/auth/.auth.env
    depends_on:
      - rabbitmq
      - postgres_auth
      - profile
    volumes:
      - .:/usr/src/app
      - /usr/src/app/node_modules
    command: npm run start:dev auth
    networks:
      - postgres_auth
      - rabbitmq

  profile:
    build:
      context: ./
      dockerfile: ./apps/profile/Dockerfile
    env_file:
      - ./apps/profile/.profile.env
    depends_on:
      - rabbitmq
      - postgres_profile
    volumes:
      - .:/usr/src/app
      - /usr/src/app/node_modules
    command: npm run start:dev profile
    networks:
      - postgres_profile
      - rabbitmq

  api:
    build:
      context: ./
      dockerfile: ./apps/api/Dockerfile
    ports:
      - '3000:3000'
    env_file:
      - .env
    depends_on:
      - rabbitmq
      - auth
      - profile
    volumes:
      - .:/usr/src/app
      - /usr/src/app/node_modules
    command: npm run start:dev api
    networks:
      - rabbitmq

networks:
  postgres_auth:
    driver: bridge
  postgres_profile:
    driver: bridge
  rabbitmq:
    driver: bridge
